def search(matrix,n,x):
	i,j=0,n-1
	while(i<n and j>=0):

		if matrix[i][j]==x:
			print ("Found at %d, %d",i ,j)
		if matrix[i][j]<x:
			i=i+1
		else:
			j=j-1
	print "Element not Found"
	return 0
#Driver Program to test above function
matrix=[{10, 20, 30, 40},
        {15, 25, 35, 45},
        {27, 29, 37, 48},
        {32, 33, 39, 50}]
search(matrix,4,33)
